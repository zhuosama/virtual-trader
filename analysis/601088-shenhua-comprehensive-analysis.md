### 3. 动态调整框架
```python
def shenhua_position_decision(coal_price_trend, market_style, dividend_yield):
    """
    中国神华仓位决策函数
    
    参数：
    - coal_price_trend: 煤价趋势（-1下跌, 0持平, 1上涨）
    - market_style: 市场风格（-1成长, 0均衡, 1价值）
    - dividend_yield: 当前股息率
    
    返回：建议仓位比例
    """
    base_position = 0.09  # 底仓9%
    
    # 煤价调整
    if coal_price_trend == -1:
        position *= 0.8  # 减仓20%
    elif coal_price_trend == 1:
        position *= 1.1  # 加仓10%
    
    # 风格调整
    if market_style == -1:  # 成长风格
        position *= 0.9  # 减仓10%
    elif market_style == 1:  # 价值风格
        position *= 1.1  # 加仓10%
    
    # 股息率调整
    if dividend_yield > 5.5:
        position *= 1.1  # 高股息吸引力强
    elif dividend_yield < 4.5:
        position *= 0.7  # 股息吸引力下降
    
    # 边界控制
    return max(0.05, min(0.12, position))  # 仓位5%-12%
```
